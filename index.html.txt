  <script>
    const svg = document.getElementById('diagram');
    const group = document.getElementById('diagramGroup');
    const outerColorInput = document.getElementById('outerColor');


    const sliders = [1, 2, 3, 4, 5].map(i => document.getElementById(`v${i}`));
    const labels = [1, 2, 3, 4, 5].map(i => document.getElementById(`v${i}Label`));


    const outerRadius = 360;
    const ringThickness = 90;
    const innerHole = 80;
    const innerMin = 135;
    const innerMax = 255;


    const sectorColors = ['#ff5ca8', '#e66bd6', '#8a63f6', '#41c46a', '#2ec4c7'];
    const sectorNames = ['A', 'B', 'C', 'D', 'E'];


    function polar(cx, cy, r, angle) {
      return {
        x: cx + r * Math.cos(angle),
        y: cy + r * Math.sin(angle)
      };
    }


    function ringSectorPath(cx, cy, r1, r2, a0, a1) {
      const p1 = polar(cx, cy, r2, a0);
      const p2 = polar(cx, cy, r2, a1);
      const p3 = polar(cx, cy, r1, a1);
      const p4 = polar(cx, cy, r1, a0);
      const largeArc = (a1 - a0) > Math.PI ? 1 : 0;
      return [
        `M ${p1.x.toFixed(2)} ${p1.y.toFixed(2)}`,
        `A ${r2} ${r2} 0 ${largeArc} 1 ${p2.x.toFixed(2)} ${p2.y.toFixed(2)}`,
        `L ${p3.x.toFixed(2)} ${p3.y.toFixed(2)}`,
        `A ${r1} ${r1} 0 ${largeArc} 0 ${p4.x.toFixed(2)} ${p4.y.toFixed(2)}`,
        'Z'
      ].join(' ');
    }


    function textAlongArc(cx, cy, r, a0, a1, text, size, color, weight = 500) {
      const mid = (a0 + a1) / 2;
      const pos = polar(cx, cy, r, mid);
      const rotate = (mid * 180 / Math.PI) + 90;
      const t = document.createElementNS('http://www.w3.org/2000/svg', 'text');
      t.setAttribute('x', pos.x);
      t.setAttribute('y', pos.y);
      t.setAttribute('fill', color);
      t.setAttribute('font-size', size);
      t.setAttribute('font-weight', weight);
      t.setAttribute('text-anchor', 'middle');
      t.setAttribute('dominant-baseline', 'middle');
      t.setAttribute('transform', `rotate(${rotate} ${pos.x} ${pos.y})`);
      t.textContent = text;
      return t;
    }


    function add(el, attrs = {}, parent = group) {
      const node = document.createElementNS('http://www.w3.org/2000/svg', el);
      for (const [k, v] of Object.entries(attrs)) node.setAttribute(k, v);
      parent.appendChild(node);
      return node;
    }


    function valueToRadius(v) {
      const t = (v - 1) / 4;
      return innerMin + t * (innerMax - innerMin);
    }


    function render() {
      group.innerHTML = '';
      const cx = 400, cy = 400;
      const outerColor = outerColorInput.value;
      const values = sliders.map(s => Number(s.value));


      sliders.forEach((s, i) => labels[i].textContent = s.value);


      add('circle', { cx, cy, r: outerRadius, fill: outerColor, filter: 'url(#shadow)' });


      const gap = 0.012;
      const step = (Math.PI * 2) / 5;


      for (let i = 0; i < 5; i++) {
        const a0 = -Math.PI / 2 + i * step + gap;
        const a1 = -Math.PI / 2 + (i + 1) * step - gap;
        const r2 = valueToRadius(values[i]);
        const r1 = innerHole;


        const path = add('path', {
          d: ringSectorPath(cx, cy, r1, r2, a0, a1),
          fill: sectorColors[i],
          stroke: '#ffffff',
          'stroke-width': 3
        });


        const mid = (a0 + a1) / 2;
        const txtPos = polar(cx, cy, (r1 + r2) / 2, mid);


        add('text', {
          x: txtPos.x,
          y: txtPos.y,
          fill: '#ffffff',
          'font-size': 28,
          'font-weight': 700,
          'text-anchor': 'middle',
          'dominant-baseline': 'middle',
          transform: `rotate(${mid * 180 / Math.PI + 90} ${txtPos.x} ${txtPos.y})`,
          'paint-order': 'stroke',
          'stroke': 'rgba(0,0,0,0.12)',
          'stroke-width': 2
        }).textContent = sectorNames[i];
      }


      add('circle', {
        cx, cy, r: 120,
        fill: '#fff2b3',
        stroke: '#f2d56b',
        'stroke-width': 2
      });


      add('text', {
        x: cx, y: cy - 10,
        fill: '#8e6a4a',
        'font-size': 34,
        'font-weight': 800,
        'text-anchor': 'middle'
      }).textContent = 'PATIENT';


      add('text', {
        x: cx, y: cy + 26,
        fill: '#8e6a4a',
        'font-size': 18,
        'font-weight': 600,
        'text-anchor': 'middle'
      }).textContent = 'Centre';


      add('circle', {
        cx, cy, r: outerRadius,
        fill: 'none',
        stroke: '#ffffff',
        'stroke-width': 6
      });


      const labelsOuter = [
        { text: 'Identité professionnelle', a0: -Math.PI * 0.90, a1: -Math.PI * 0.55 },
        { text: 'Auto-réflexive', a0: -Math.PI * 0.05, a1: Math.PI * 0.30 },
        { text: 'Identité personnelle', a0: Math.PI * 0.72, a1: Math.PI * 1.05 },
        { text: 'Praticien', a0: Math.PI * 0.38, a1: Math.PI * 0.55 }
      ];


      labelsOuter.forEach(item => {
        const t = textAlongArc(cx, cy, 320, item.a0, item.a1, item.text, 24, '#ffffff', 500);
        group.appendChild(t);
      });
    }


    function downloadSVG() {
      const clone = svg.cloneNode(true);
      clone.setAttribute('xmlns', 'http://www.w3.org/2000/svg');
      const serializer = new XMLSerializer();
      const source = serializer.serializeToString(clone);
      const blob = new Blob([source], { type: 'image/svg+xml;charset=utf-8' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'diagramme-radial.svg';
      document.body.appendChild(a);
      a.click();
      a.remove();
      URL.revokeObjectURL(url);
    }


    function copyConfig() {
      const config = {
        outerColor: outerColorInput.value,
        values: sliders.map(s => Number(s.value))
      };
      navigator.clipboard.writeText(JSON.stringify(config, null, 2));
      alert('Configuration copiée dans le presse-papiers.');
    }


    sliders.forEach(s => s.addEventListener('input', render));
    outerColorInput.addEventListener('input', render);
    document.getElementById('downloadSvg').addEventListener('click', downloadSVG);
    document.getElementById('copyConfig').addEventListener('click', copyConfig);


    render();
  </script>
